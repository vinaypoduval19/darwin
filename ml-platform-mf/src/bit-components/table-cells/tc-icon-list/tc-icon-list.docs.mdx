import {TcIconList} from './tc-icon-list'

A component that does renders an icon list.

### Component usage

```js
import {TcIconList} from '@sporta-technology/d11-web-components.table-cells.tc-icon-list'

<TcIconList
  size={TableCellSize.Large}
  onClick={handleOnClick}
  iconList={iconListMockData}
/>
```
